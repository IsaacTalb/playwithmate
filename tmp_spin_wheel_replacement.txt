import { useState } from 'react'
import AdSlot from '../../src/components/AdSlot'
const OPTIONS = [
    'Truth',
    'Dare, "The tallest one from group can dare."',
    'Quiz',
    'Mystery',
    'Sing a song',
    'Dance for 1 minute',
    'Share an embarrassing story',
    'Do 10 push-ups',
    'Imitate someone in the group',
    'Tell a joke'
]


export default function SpinWheel(){
const [result, setResult] = useState(null)
const spin = () => {
const pick = OPTIONS[Math.floor(Math.random()*OPTIONS.length)]
setResult(pick)
}
return (
    <div className="p-4 max-w-xl mx-auto">
        <div className="flex flex-row items-center justify-between mb-6">
            <h1 className="text-2xl font-bold">Spin the Wheel</h1>
            <a href="/" className="flex items-center gap-2 px-4 py-2 border-2 border-blue-500 text-blue-500 rounded hover:bg-blue-50">
            <span>üè†</span>
            Back to Home
            </a>
        </div>
        <div className="mt-4 flex gap-2">
            <button onClick={spin} className="flex-1 p-3 bg-yellow-500 rounded">Spin</button>
        </div>
        {result && <div className="mt-4 bg-white p-4 rounded">Result: {result}</div>}
        <div className="mt-4"><AdSlot /></div>
    </div>
    )
}
